### Manual MakeFile
```make
all: static_block dynamic_block
block.o: source/block.c
    cc -fPIC -c source/block.c -o block.o
block.so: source/block.c
    cc -shared -o block.o source/block.c
static_block: block.o
    cc block.o program.c -o static_block
dynamic_block: block.so
    cc block.o program.c -o dynamic_block -Wl,-rpath='$$ORIGIN'
clean:
    rm *.o *.so dynamic_block static_block
```
